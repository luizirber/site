Funções

  • Até agora colocamos o código no arquivo, e toda vez que ele é rodado tudo é executado em sequência.
  • Vamos definir uma função que lê os dados de um arquivo e retorna um dicionário com os dados.
def le_dados(arquivo):
    pessoas = {}
    fp = open(arquivo, 'r')
    for line in fp:
        pessoa, lat, lon = line[:-1].split()
        pessoas[pessoa] = (float(lat), float(lon))
    return pessoas

le_dados(sys.argv[1])

Calculando distância entre duas pessoas

  • Para resolver o problema, precisamos calcular a distância entre duas pessoas
import math

def distance(p1, p2):
    dlat_squared = (p1[0] - p2[0]) ** 2
    dlon_squared = (p1[1] - p2[1]) ** 2
    return math.sqrt(dlat_squared + dlon_squared)

class Pessoa(object):

    def __init__(self, nome, lat, lon):
        self.nome = nome
        self.lat = lat
        self.lon = lon

    def distancia(self, p2):
        dlat_squared = (self.lat - p2.lat) ** 2
        dlon_squared = (self.lon - p2.lon) ** 2
        return math.sqrt(dlat_squared + dlon_squared)

    def imprime(self):
        dados = tuple([d[1].nome for d in self.distancias][:3])
        return " ".join([self.nome, "%s,%s,%s\n" % dados])
